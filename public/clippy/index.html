Implementation Example 
clippy.load('Clippy', function(agent) {
    agent.show();
    agent.setRecommendedVoice();

    // New reusable Ask feature
    agent.ask({
        title: 'User Prompt',
        onAsk: (val) => agent.speak('Received: ' + val)
    });
});
